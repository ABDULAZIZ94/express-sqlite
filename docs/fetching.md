    function post_data (data){
      fetch("/create", {
        method:'POST',
        headers: {
          'Content-Type': 'application/x-www-form-urlencoded',
        },
        body: data
      })
      .then(response => {
          const contentType = response.headers.get('content-type');
          console.log(contentType);
          return response.text();
      })
      .then(t => {
          alert('replied: '+t);
      })
      .catch(error => console.error(error));
    }
